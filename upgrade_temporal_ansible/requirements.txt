# Temporal SDK (the worker + client)
temporalio>=1.7

# Ansible — the worker shells out to `ansible-playbook` to run the bundled
# ansible/playbooks set. PyYAML (used to generate the inventory) ships as an
# Ansible dependency.
ansible-core>=2.16.6

# The Junos playbooks use the juniper.device collection (PyEZ based). Install
# it once with:  ansible-galaxy collection install -r ansible/requirements.yml
#
# The collection is Python-driven by Juniper PyEZ (junos-eznc), which provides
# the NETCONF stack (ncclient). Without these the modules fail at runtime with
# "ncclient module could not be imported."
junos-eznc>=2.6.7
ncclient>=0.6.13
lxml>=4.6.0

# juniper.device 1.0.7 imports `LooseVersion` from the `looseversion` package
# (`from looseversion import LooseVersion`). It is NOT a hard dependency of the
# collection, so on a clean venv the import fails silently
# (HAS_LOOSE_VERSION = False) and the modules crash with
# "NameError: name 'LooseVersion' is not defined". Install it explicitly.
looseversion>=1.3.0

# jxmlease is required by the juniper.device modules (XML<->dict handling) and,
# like looseversion, is not pulled in automatically. Without it the modules
# fail with "jxmlease >= 1.0.1 is required for this module".
jxmlease>=1.0.3
